```swift
protocol EchoAIProviding: Sendable {
    func respond(messages: [EchoChatMessage]) async throws -> String
    func summarize(prompt: String) async throws -> String
    func generateProfile(prompt: String) async throws -> String
    var isAvailable: Bool { get }
}

enum EchoChatRole: String, Codable, Sendable {
    case system
    case user
    case assistant
}

struct EchoChatMessage: Codable, Sendable, Identifiable {
    let id: UUID
    let role: EchoChatRole
    let content: String
    let createdAt: Date
}

enum EchoAIError: Error, LocalizedError, Sendable {
    case providerUnavailable(String)
    case networkError(Error)
    case invalidResponse
    case rateLimited
    case apiKeyMissing
    case modelNotSupported
}
```

```swift
import Foundation
import AIProxy

/// Pro-tier AI provider using Claude Haiku 4.5 via AIProxy.
///
/// The partial key is safe to include in the binary — AIProxy's split-key model
/// ensures the full API key never touches the device.
///
/// Set AIPROXY_DEVICE_CHECK_BYPASS environment variable in the Xcode scheme
/// for simulator builds (value obtained from AIProxy dashboard).
final class AnthropicAIProvider: EchoAIProviding, @unchecked Sendable {

    // MARK: - AIProxy Configuration
    // Partial key from AIProxy dashboard — safe to commit to source control.
    // Full key is stored server-side by AIProxy and never reaches the device.
    private static let partialKey = "REPLACE_WITH_AIPROXY_PARTIAL_KEY"
    private static let serviceURL = "REPLACE_WITH_AIPROXY_SERVICE_URL"

    private static let model = "claude-haiku-4-5-20251001"

    private let service: AnthropicService

    init() {
        self.service = AIProxy.anthropicService(
            partialKey: Self.partialKey,
            serviceURL: Self.serviceURL
        )
    }

    // MARK: - EchoAIProviding

    var isAvailable: Bool { true }

    func summarize(prompt: String) async throws -> String {
        let requestBody = AnthropicMessageRequestBody(
            maxTokens: 512,
            messages: [AnthropicMessageParam(content: prompt, role: .user)],
            model: Self.model,
            system: "你是一个生活数据分析助手。根据提供的数据，生成简洁的中文摘要，描述用户今天的生活，不评价，不建议。"
        )
        let response = try await service.messageRequest(body: requestBody, secondsToWait: 30)
        guard let textBlock = response.content.first,
              case let .textBlock(block) = textBlock else {
            throw EchoAIError.invalidResponse
        }
        return block.text
    }

    func respond(messages: [EchoChatMessage]) async throws -> String {
        let anthropicMessages = messages
            .filter { $0.role != .system }
            .map { msg -> AnthropicMessageParam in
                let role: AnthropicMessageParam.Role = msg.role == .user ? .user : .assistant
                return AnthropicMessageParam(content: msg.content, role: role)
            }
        let systemMsg = messages.first(where: { $0.role == .system })?.content ?? ""

        let requestBody = AnthropicMessageRequestBody(
            maxTokens: 1024,
            messages: anthropicMessages,
            model: Self.model,
            system: systemMsg
        )
        let response = try await service.messageRequest(body: requestBody, secondsToWait: 30)
        guard let textBlock = response.content.first,
              case let .textBlock(block) = textBlock else {
            throw EchoAIError.invalidResponse
        }
        return block.text
    }

    func generateProfile(prompt: String) async throws -> String {
        return try await summarize(prompt: prompt)
    }
}
```

IMPORTANT: If AIProxySwift's actual type names differ from the above (e.g., the `Role` enum or `textBlock` pattern), use Xcode's autocomplete to discover the correct names from the resolved package source. The pattern will be structurally identical — only the exact Swift type names may vary slightly. Do NOT use URLSession to call Anthropic directly — the entire purpose is routing through AIProxy.
